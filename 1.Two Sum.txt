/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
    var answer = [];
    for(var i = 0; i < nums.length-1; i++){
        var value = 0;
        for(var j = i+1; j < nums.length; j++){
            value = nums[i] + nums[j];
            if(value == target){
                answer.push(i);
                answer.push(j);
            }
        }
    }
    return answer;
};
