/**
 * @param {number} x
 * @return {boolean}
 */
var isPalindrome = function(x) {
    var str = x.toString();
    if(str.charAt(0) == '-'){
        return false;
    }
    
    var length = str.length-1;
    for(var i = 0; i < str.length/2; i++){
        if(str.charAt(i) != str.charAt(length-i)){
            return false;
        }
    }
    
    return true;
    //return x.toString() === x.toString().split('').reverse().join('')
};
