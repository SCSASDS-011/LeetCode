/**
 * @param {number[]} nums
 * @return {number[]}
 */
var findDuplicates = function(nums) {
    var set = new Set();
    var arr = [];
    for(var i = 0; i < nums.length; i++){
        if(!set.has(nums[i])){
            set.add(nums[i]);
        }else{
            arr.push(nums[i]);
        }
    }
    return arr;
};
