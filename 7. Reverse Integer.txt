/**
 * @param {number} n
 * @return {string[]}
 */
var fizzBuzz = function(n) {
    var answer = [];
    
    for(var i = 1; i <= n; i++){
        if(i % 3 == 0 && i % 5 != 0){
            answer.push('Fizz');
        }else if(i % 3 != 0 && i % 5 == 0){
            answer.push('Buzz');
        }else if(i % 3 == 0 && i % 5 == 0){
            answer.push('FizzBuzz');
        }else{
            answer.push(i+"");
        }
    }
    return answer;
};
